% cover.tex — full cover spread for a 6x9 in paperback at Lulu.
% Cover dimensions depend on spine width, which depends on page count and
% paper stock. Lulu publishes a cover-template calculator; once the final
% page count is locked, set TRIMW, TRIMH, SPINE, and BLEED below.
%
% Stock assumption for v1: Lulu cream uncoated, 60# (book paper).
% At ~320 pages, the spine will be roughly 0.74 in. Refresh this before
% submitting.
%
% Build separately with:  xelatex cover.tex
% This is intentionally not included from main.tex; Lulu wants two PDFs
% (interior + cover) and merging them at the LaTeX layer adds no value.

\documentclass[12pt]{article}

\usepackage{fontspec}
\usepackage{xcolor}
\usepackage[absolute, overlay]{textpos}
\usepackage{graphicx}

\defaultfontfeatures{Path=../fonts/, Ligatures=TeX}

    ItalicFont  = *-Italic.otf
]
\newfontfamily\monofont{JetBrainsMono}[
    UprightFont = *-Regular.ttf,
    Scale=0.86
]

\definecolor{paper}{HTML}{F5EFE0}

\def\BLEED{0.125in}

\newlength{\fullW}\setlength{\fullW}{\dimexpr 2\TRIMW + \SPINE + 2\BLEED\relax}
\newlength{\fullH}\setlength{\fullH}{\dimexpr \TRIMH + 2\BLEED\relax}

\usepackage[
    paperwidth=\fullW,
    paperheight=\fullH,
    margin=0in,
    nohead, nofoot
]{geometry}

\pagecolor{paper}
\pagestyle{empty}

\setlength{\TPHorizModule}{1in}
\setlength{\TPVertModule}{1in}

% Layout coordinates (from top-left, in inches, relative to the bleed page).
\newlength{\backX}  \setlength{\backX}{\BLEED}
\newlength{\spineX} \setlength{\spineX}{\dimexpr \BLEED + \TRIMW\relax}
\newlength{\frontX} \setlength{\frontX}{\dimexpr \BLEED + \TRIMW + \SPINE\relax}
